1. Executive Summary
Problem Statement
73% of qualified candidates fail job interviews due to poor body language, filler words, and lack of confidence - NOT lack of skills. Students and professionals have no way to practice with real-time feedback before high-stakes interviews.
Solution
InterviewIQ is a real-time AI interview coach that watches your posture, listens to your speech, analyzes your answers, and provides instant feedback to help you improve. Think of it as having a personal interview coach available 24/7 at a fraction of the cost.
Unique Value Propositions

Multi-modal Analysis - Combines video (body language) + audio (speech patterns) + text (answer quality)
Real-time Feedback - Sub-500ms latency using Vision Agents SDK
Invisible Mode - Silent text prompts that won't be detected during real interviews
Before/After Proof - Measurable improvement metrics
Enterprise-Ready - Stress detection, confidence scoring, PDF reports

Target Market

Primary: College students (placement preparation)
Secondary: Working professionals (job transitions)
Tertiary: Universities (career services departments)

Market Size:

20M+ students graduate annually in India
150M+ job applications in US yearly
$5B+ interview prep market

  ✓ Pause Patterns
    - Average pause length
    - Awkward silence detection (>3 seconds)
    - Natural pause score
  
  ✓ Tone Analysis
    - Pitch variation (confidence indicator)
    - Volume consistency
    - Energy level tracking

# Output:
- Live transcript with highlighted filler words
- Speech pace meter
- Tone confidence score (0-100)

D. Real-time Feedback System
python# Two Modes:

1. Practice Mode (Voice + Text Feedback)
   - Full audio coaching during pauses
   - Text overlays on screen
   - Detailed metrics displayed
   - Slower paced for learning

2. Invisible Mode (Text Only)
   - Silent text prompts
   - Minimal UI distraction
   - Can run on second screen/phone
   - Undetectable by interviewer

# Feedback Triggers:
- Posture drops below 60 → "Sit up straight"
- 3+ filler words in 30 sec → "Slow down, breathe"
- No eye contact for 10 sec → "Look at camera"
- Silence >5 seconds → "Take your time, structure your answer"
- Low energy detected → "Show more enthusiasm"

# Delivery:
- Text: Instant display (50ms)
- Voice: ElevenLabs TTS (200ms latency)
- Priority queue (urgent alerts first)

11.1 Main Agent Implementation
python# backend/agents/interview_agent.py

from vision_agents import Agent
from vision_agents.integrations import getstream, gemini, deepgram, elevenlabs
from processors.body_language_processor import BodyLanguageProcessor
from processors.speech_analysis_processor import SpeechAnalysisProcessor
from processors.state_management_processor import StateManagementProcessor
from services.feedback_service import FeedbackService

class InterviewAgent:
    """
    Main interview agent that coordinates multi-modal analysis.
    """
    
    def __init__(
        self,
        user_id: str,
        session_id: str,
        mode: str = "practice",  # "practice" or "invisible"
        question_bank: list = None
    ):
        self.user_id = user_id
        self.session_id = session_id
        self.mode = mode
        self.question_bank = question_bank or self._load_default_questions()
        self.feedback_service = FeedbackService()
        
        # Initialize processors
        self.body_processor = BodyLanguageProcessor(
            model_path="models/yolo11n-pose.pt",
            fps=5,  # Process 5 frames per second
            device="cuda" if torch.cuda.is_available() else "cpu"
        )
        
        self.speech_processor = SpeechAnalysisProcessor(
            filler_words=["umm", "uh", "like", "you know", "basically"],
            target_wpm=140,  # Ideal words per minute
            alert_threshold=3  # Alert after 3 filler words
        )
        
        self.state_processor = StateManagementProcessor(
            session_id=session_id
        )
        
        # Create Vision Agent
        self.agent = Agent(
            edge=getstream.Edge(),
            agent_user=self._create_agent_user(),
            instructions=self._load_instructions(),
            llm=gemini.LLM("gemini-2.0-flash-exp"),
            tts=elevenlabs.TTS(voice_id="coach_voice") if mode == "practice" else None,
            stt=deepgram.STT(model="nova-2"),
            processors=[
                self.body_processor,
                self.speech_processor,
                self.state_processor
            ]
        )
        
        # Setup event handlers
        self._setup_event_handlers()
    
    def _create_agent_user(self):
        """Create agent user object"""
        return {
            "id": "interview_coach",
            "name": "Interview Coach",
            "role": "AI Coach"
        }
    
    def _load_instructions(self):
        """Load agent instructions based on mode"""
        if self.mode == "practice":
            return """
            You are an expert interview coach helping candidates improve.
            
            Your role:
            1. Ask interview questions one at a time
            2. Listen carefully to answers
            3. Analyze body language, speech patterns, and content
            4. Provide constructive, specific feedback
            5. Be encouraging but honest
            
            Feedback style:
            - Be positive and supportive
            - Give specific, actionable advice
            - Celebrate improvements
            - Point out both strengths and areas to improve
            
            When you notice issues:
            - Poor posture → "Remember to sit up straight"
            - Too many filler words → "Take a breath, slow down"
            - No eye contact → "Try looking at the camera"
            - Long pauses → "Take your time to think"
            
            After each answer, provide:
            1. One strength
            2. One area to improve
            3. Suggested rewording (if needed)
            """
        else:  # invisible mode
            return """
            You are a silent interview assistant.
            
            Provide only text-based prompts that help the candidate.
            Never speak out loud (no TTS).
            
            Your prompts should be:
            - Brief (5-10 words max)
            - Actionable
            - Encouraging
            - Silent (text only)
            
            Examples:
            - "Mention the project metrics"
            - "Give a specific example"
            - "Slow down"
            - "Good! Keep going"
            """
    
    def _setup_event_handlers(self):
        """Setup event handlers for real-time feedback"""
        
        @self.agent.on_event("body_language_update")
        async def handle_body_language(data):
            """Handle body language updates"""
            posture_score = data.get("posture_score", 0)
            eye_contact = data.get("eye_contact_percentage", 0)
            
            # Generate feedback if scores are low
            if posture_score < 60:
                await self.feedback_service.send_feedback(
                    session_id=self.session_id,
                    type="text",
                    message="💪 Sit up straight",
                    priority="high"
                )
            
            if eye_contact < 50:
                await self.feedback_service.send_feedback(
                    session_id=self.session_id,
                    type="text",
                    message="👀 Look at the camera",
                    priority="medium"
                )
        
        @self.agent.on_event("speech_analysis_update")
        async def handle_speech(data):
            """Handle speech analysis updates"""
            filler_count = data.get("filler_words_count", 0)
            speech_pace = data.get("words_per_minute", 0)
            
            # Alert if too many filler words
            if filler_count >= 3:
                await self.feedback_service.send_feedback(
                    session_id=self.session_id,
                    type="text" if self.mode == "invisible" else "voice",
                    message="Take a breath. Slow down.",
                    priority="high"
                )
            
            # Alert if speaking too fast/slow
            if speech_pace > 180:
                await self.feedback_service.send_feedback(
                    session_id=self.session_id,
                    type="text",
                    message="🗣️ Speaking too fast",
                    priority="medium"
                )
        
        @self.agent.on_event("answer_complete")
        async def handle_answer_complete(data):
            """Handle when user finishes answering"""
            # Get LLM analysis
            analysis = await self._analyze_answer(data)
            
            # Send comprehensive feedback
            await self.feedback_service.send_feedback(
                session_id=self.session_id,
                type="voice" if self.mode == "practice" else "text",
                message=analysis["feedback"],
                priority="normal"
            )
            
            # Update session metrics
            await self.state_processor.update_metrics(
                session_id=self.session_id,
                metrics=analysis["metrics"]
            )
    
    async def _analyze_answer(self, data):
        """Use LLM to analyze answer quality"""
        prompt = f"""
        Question asked: {data['question']}
        Candidate's answer: {data['transcript']}
        
        Body language metrics:
        - Posture: {data['posture_score']}/100
        - Eye contact: {data['eye_contact']}/100
        
        Speech metrics:
        - Filler words: {data['filler_words']}
        - Speech pace: {data['wpm']} WPM
        
        Provide:
        1. Content score (0-100)
        2. One key strength
        3. One improvement area
        4. Suggested answer rewrite
        """
        
        response = await self.agent.llm.generate(prompt)
        
        return {
            "feedback": response["feedback"],
            "metrics": response["metrics"]
        }
    
    def _load_default_questions(self):
        """Load default interview questions"""
        return [
            "Tell me about yourself.",

            "What are your greatest strengths?",
            "Describe a challenge you faced and how you overcame it.",
            "Where do you see yourself in 5 years?"
        ]
    
    async def start_session(self):
        """Start interview session"""
        await self.agent.connect()

python# backend/processors/body_language_processor.py

import cv2
import numpy as np
from ultralytics import YOLO
from vision_agents.processors import Processor

class BodyLanguageProcessor(Processor):
    """
    Processes video frames to analyze body language using YOLO pose detection.
    """
    
    def __init__(self, model_path: str, fps: int = 5, device: str = "cuda"):
        super().__init__()
        self.model = YOLO(model_path)
        self.fps = fps
        self.device = device
        self.frame_count = 0
        self.metrics_history = []
        
        # Thresholds
        self.GOOD_POSTURE_THRESHOLD = 75
        self.EYE_CONTACT_THRESHOLD = 70
        
    async def process_frame(self, frame: np.ndarray) -> dict:
        """Process single video frame"""
        self.frame_count += 1
        
        # Only process every Nth frame (based on FPS)
        if self.frame_count % (30 // self.fps) != 0:
            return None
        
        # Run YOLO inference
        results = self.model(frame, device=self.device)[0]
        
        if len(results.keypoints) == 0:
            return None
        
        # Extract keypoints
        keypoints = results.keypoints.xy[0].cpu().numpy()
        
        # Calculate metrics
        metrics = {
            "posture_score": self._calculate_posture_score(keypoints),
            "eye_contact_percentage": self._calculate_eye_contact(keypoints),
            "fidgeting_detected": self._detect_fidgeting(keypoints),
            "hand_gestures": self._analyze_hand_gestures(keypoints),
            "timestamp": self.frame_count / 30  # Assuming 30 FPS
        }
        
        # Store in history
        self.metrics_history.append(metrics)
        
        # Emit event for real-time feedback
        await self.emit_event("body_language_update", metrics)
        
        return metrics
    
    def _calculate_posture_score(self, keypoints: np.ndarray) -> float:
        """
        Calculate posture score (0-100) based on shoulder/back alignment.
        
        Good posture indicators:
        - Level shoulders
        - Straight back
        - Head not tilted too much
        """
        # YOLO keypoints indices
        LEFT_SHOULDER = 5
        RIGHT_SHOULDER = 6
        LEFT_HIP = 11
        RIGHT_HIP = 12
        NOSE = 0
        
        try:
            # Get coordinates
            left_shoulder = keypoints[LEFT_SHOULDER]
            right_shoulder = keypoints[RIGHT_SHOULDER]
            left_hip = keypoints[LEFT_HIP]
            right_hip = keypoints[RIGHT_HIP]
            nose = keypoints[NOSE]
            
            # Calculate shoulder alignment (should be level)
            shoulder_diff = abs(left_shoulder[1] - right_shoulder[1])
            max_diff = 50  # pixels
            shoulder_score = max(0, 100 - (shoulder_diff / max_diff * 100))
            
            # Calculate back straightness
            shoulder_mid = (left_shoulder + right_shoulder) / 2
            hip_mid = (left_hip + right_hip) / 2
            back_angle = np.arctan2(
                shoulder_mid[0] - hip_mid[0],
                shoulder_mid[1] - hip_mid[1]
            )
            back_score = max(0, 100 - abs(np.degrees(back_angle)))
            
            # Calculate head tilt
            head_tilt = abs(nose[0] - shoulder_mid[0])
            head_score = max(0, 100 - (head_tilt / 100 * 100))
            
            # Weighted average
            posture_score = (
                shoulder_score * 0.4 +
                back_score * 0.4 +
                head_score * 0.2
            )
            
            return round(posture_score, 2)
        
        except Exception as e:
            return 50.0  # Default neutral score
    
    def _calculate_eye_contact(self, keypoints: np.ndarray) -> float:
        """
        Calculate eye contact percentage based on face direction.
        """
        # Simplified: Check if nose is centered in frame
        NOSE = 0
        LEFT_EYE = 1
        RIGHT_EYE = 2
        
        try:
            nose = keypoints[NOSE]
            left_eye = keypoints[LEFT_EYE]
            right_eye = keypoints[RIGHT_EYE]
            
            # Get frame dimensions (assume standard)
            frame_center_x = 640 / 2
            
            # Calculate how centered the face is
            face_center_x = (left_eye[0] + right_eye[0]) / 2
            deviation = abs(face_center_x - frame_center_x)
            
            # Eye contact score (higher = more centered)
            eye_contact = max(0, 100 - (deviation / 200 * 100))
            
            return round(eye_contact, 2)
        
        except Exception as e:
            return 50.0
    
    def _detect_fidgeting(self, keypoints: np.ndarray) -> bool:
        """
        Detect fidgeting based on movement variance.
        """
        if len(self.metrics_history) < 10:
            return False
        
        # Get last 10 frames of posture scores
        recent_scores = [m["posture_score"] for m in self.metrics_history[-10:]]
        
        # Calculate variance
        variance = np.var(recent_scores)
        
        # High variance = fidgeting
        return variance > 100
    
    def _analyze_hand_gestures(self, keypoints: np.ndarray) -> dict:
        """
        Analyze hand gestures.
        """
        LEFT_WRIST = 9
        RIGHT_WRIST = 10
        
        try:
            left_wrist = keypoints[LEFT_WRIST]
            right_wrist = keypoints[RIGHT_WRIST]
            
            # Check if hands are visible
            hands_visible = (
                left_wrist[0] > 0 and left_wrist[1] > 0 and
                right_wrist[0] > 0 and right_wrist[1] > 0
            )
            
            # Check if hands are near face (nervous gesture)
            nose = keypoints[0]
            left_near_face = np.linalg.norm(left_wrist - nose) < 150
            right_near_face = np.linalg.norm(right_wrist - nose) < 150
            
            return {
                "hands_visible": hands_visible,
                "touching_face": left_near_face or right_near_face,
                "gesture_score": 75 if hands_visible and not (left_near_face or right_near_face) else 50
            }
        
        except Exception as e:
            return {
                "hands_visible": False,
                "touching_face": False,
                "gesture_score": 50
            }
    
    async def get_session_summary(self) -> dict:
        """Get summary of entire session"""
        if not self.metrics_history:
            return {}
        
        return {
            "average_posture": np.mean([m["posture_score"] for m in self.metrics_history]),
            "average_eye_contact": np.mean([m["eye_contact_percentage"] for m in self.metrics_history]),
            "fidgeting_incidents": sum([m["fidgeting_detected"] for m in self.metrics_history]),
            "total_frames_processed": len(self.metrics_history)
        }
